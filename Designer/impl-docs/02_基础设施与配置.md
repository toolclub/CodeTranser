## 2.3 日志(structlog + stdlib)

```python
# app/infra/logging.py
import logging
import sys
import structlog
from app.config import Settings
from app.infra.tracing import trace_id_ctx

def _add_trace_id(logger, method_name, event_dict):
    tid = trace_id_ctx.get(None)
    if tid: event_dict["trace_id"] = tid
    return event_dict

def _redact(logger, method_name, event_dict):
    SENS = {"authorization", "cookie", "password", "token", "api_key"}
    for k in list(event_dict):
        if k.lower() in SENS: event_dict[k] = "***"
    return event_dict

def configure_logging(settings: Settings) -> None:
    level = getattr(logging, settings.LOG_LEVEL.upper(), logging.INFO)

    logging.basicConfig(
        format="%(message)s",
        stream=sys.stdout,
        level=level,
    )

    structlog.configure(
        processors=[
            structlog.contextvars.merge_contextvars,
            _add_trace_id,
            _redact,
            structlog.processors.add_log_level,
            structlog.processors.TimeStamper(fmt="iso", utc=True),
            structlog.processors.StackInfoRenderer(),
            structlog.processors.format_exc_info,
            structlog.processors.JSONRenderer(),
        ],
        wrapper_class=structlog.make_filtering_bound_logger(level),
        cache_logger_on_first_use=True,
    )

def get_logger(name: str):
    return structlog.get_logger(name)
```

```python
from app.infra.tracing import trace_id_ctx, new_trace_id

HEADER = "X-Trace-Id"

class TraceIdMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        tid = request.headers.get(HEADER) or new_trace_id()
        token = trace_id_ctx.set(tid)
        try:
            response = await call_next(request)
            response.headers[HEADER] = tid
            return response
```

```python
# app/middlewares/error_handler.py
from fastapi import FastAPI, Request
from fastapi.responses import JSONResponse
from pydantic import ValidationError as PydanticValidationError
from app.domain.errors import DomainError
from app.infra.tracing import current_trace_id
from app.infra.logging import get_logger

log = get_logger(__name__)

def _resp(code: str, message: str, status: int, data=None):
    return JSONResponse(
        status_code=status,
        content={"code": 1, "error_code": code, "message": message, "data": data, "trace_id": current_trace_id()},
    )

def register(app: FastAPI) -> None:
    @app.exception_handler(DomainError)
    async def _domain(req: Request, exc: DomainError):
        return _resp(exc.code, exc.message, exc.http_status, data=exc.extra)

    @app.exception_handler(PydanticValidationError)
    async def _pyd(req: Request, exc: PydanticValidationError):
        return _resp("VALIDATION_FAILED", "validation error", 422, data=exc.errors())

    @app.exception_handler(Exception)
    async def _any(req: Request, exc: Exception):
        log.exception("unhandled_exception")
        return _resp("INTERNAL_UNEXPECTED", "internal error", 500)
```

```python
from typing import AsyncIterator
from sqlalchemy.ext.asyncio import create_async_engine, async_sessionmaker, AsyncSession
from app.config import Settings

def create_engine(settings: Settings):

        settings.DATABASE_URL,

        echo=False,
    )

def create_session_factory(engine) -> async_sessionmaker[AsyncSession]:
    return async_sessionmaker(engine, expire_on_commit=False, class_=AsyncSession)

@asynccontextmanager
async def session_scope(factory: async_sessionmaker[AsyncSession]) -> AsyncIterator[AsyncSession]:
    async with factory() as s:
        try:
            yield s
            await s.commit()
        except Exception:
            await s.rollback()
            raise
```

### 2.10.2 UserRepo 示例(其余 Repo 同样模板,本章只给 User / AppSetting / Audit 作为基础设施)

```python
# app/repositories/user_repo.py
from abc import ABC, abstractmethod
from sqlalchemy import select
from app.models.mysql.user import User
from app.repositories.base import SqlRepoBase
from app.domain.errors import NotFound

class UserRepo(ABC):
    @abstractmethod
    async def get_by_external(self, external_id: str) -> User: ...
    @abstractmethod
    async def upsert_from_sso(self, external_id: str, username: str,
                              display_name: str, email: str, is_admin: bool) -> User: ...

class SqlUserRepo(SqlRepoBase, UserRepo):
    async def get_by_external(self, external_id: str) -> User:
        r = await self._s.execute(select(User).where(User.external_id == external_id))
        u = r.scalar_one_or_none()
        if u is None: raise NotFound(f"user {external_id}")
        return u

    async def upsert_from_sso(self, external_id, username, display_name, email, is_admin):
        r = await self._s.execute(select(User).where(User.external_id == external_id))
        u = r.scalar_one_or_none()
        if u is None:
            u = User(external_id=external_id, username=username, display_name=display_name,
                    email=email, is_admin=is_admin)
            self._s.add(u); await self._s.flush()
        else:
            u.username = username; u.display_name = display_name
            u.email = email; u.is_admin = is_admin
        return u
```

### 2.10.5 SettingsService(10s 本地缓存 + pub/sub)

```python
# app/settings_service.py
import asyncio
import json
import time
from redis.asyncio import Redis
from app.infra.db.session import session_scope
from app.repositories.app_setting_repo import AppSettingRepo

class SettingsService:
    CHANNEL = "app_setting:invalidate"
    TTL = 10.0

    def __init__(self, session_factory, redis: Redis):
        self._sf = session_factory
        self._redis = redis
        self._cache: dict[str, tuple[dict, float]] = {}

    async def start(self) -> None:
        pubsub = self._redis.pubsub()
        await pubsub.subscribe(self.CHANNEL)
        asyncio.create_task(self._listen(pubsub))

    async def _listen(self, pubsub):
        async for msg in pubsub.listen():
            if msg.get("type") == "message":
                key = msg.get("data")
                self._cache.pop(key, None)

    async def get(self, key: str, default=None):
        v = self._cache.get(key)
        if v and (time.time() - v[1]) < self.TTL:
            return v[0].get("v", default)
        async with session_scope(self._sf) as s:
            repo = AppSettingRepo(s)
            raw = await repo.get(key) or {"v": default}
        self._cache[key] = (raw, time.time())
        return raw.get("v", default)

    async def set(self, key: str, value, user_id: int, note: str = ""):
        async with session_scope(self._sf) as s:
            await AppSettingRepo(s).set(key, {"v": value}, user_id, note)
        await self._redis.publish(self.CHANNEL, key)
        self._cache.pop(key, None)
```

## 2.11 AuthMiddleware(SSO token 校验框架)

> v1 对接公司 SSO,具体 token 解析放 `middlewares/sso_adapter.py`。本章先给接口和 stub,完整实现可在 10 章按 SSO 协议补齐。

```python
# app/middlewares/auth.py
from dataclasses import dataclass
from fastapi import Request
from starlette.middleware.base import BaseHTTPMiddleware
from app.domain.errors import Unauthorized
from app.infra.db.session import session_scope
from app.repositories.user_repo import SqlUserRepo
from app.config import get_settings

@dataclass(frozen=True, slots=True)
class CurrentUser:
    id: int
    external_id: str
    username: str
    display_name: str
    email: str
    is_admin: bool
    roles: tuple[str, ...] = ()

PUBLIC_PATHS = {"/healthz", "/readyz", "/metrics", "/docs", "/openapi.json", "/redoc"}

class AuthMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        if request.url.path in PUBLIC_PATHS or request.url.path.startswith("/public/"):
            return await call_next(request)

        token = self._extract(request)
        if not token:
            raise Unauthorized("missing token")

        claims = await self._verify(token)        # SSO adapter
        user = await self._load_user(request.app.state.container, claims)
        request.state.user = user
        return await call_next(request)

    def _extract(self, req: Request) -> str | None:
        h = req.headers.get("authorization", "")
        return h[7:] if h.startswith("Bearer ") else None

    async def _verify(self, token: str) -> dict:
        from app.middlewares.sso_adapter import verify_token
        return await verify_token(token)

    async def _load_user(self, container, claims: dict) -> CurrentUser:
        async with session_scope(container.session_factory) as s:
            repo = SqlUserRepo(s)
            u = await repo.upsert_from_sso(
                external_id=claims["sub"],
                username=claims.get("preferred_username", claims["sub"]),
                display_name=claims.get("name", claims["sub"]),
                email=claims.get("email", ""),
                is_admin=(claims["sub"] in get_settings().ADMIN_EXTERNAL_IDS)
                         or ("admin" in claims.get("roles", [])),
            )
        return CurrentUser(
            id=u.id, external_id=u.external_id, username=u.username,
            display_name=u.display_name, email=u.email, is_admin=u.is_admin,
            roles=tuple(claims.get("roles", [])),
        )

# Depends:
from fastapi import Request, HTTPException

def require_user(request: Request) -> CurrentUser:
    u = getattr(request.state, "user", None)
    if u is None: raise HTTPException(401)
    return u

def require_admin(request: Request) -> CurrentUser:
    u = require_user(request)
    if not u.is_admin: raise HTTPException(403)
    return u
```

## 2.12 RateLimit 中间件(基于 Redis)

```python
# app/middlewares/rate_limit.py
from fastapi import Request
from starlette.middleware.base import BaseHTTPMiddleware
from app.domain.errors import TooManyRequests

class RateLimitMiddleware(BaseHTTPMiddleware):
    """基于 Redis 的滑动窗口限流。按 (user_id, action) 分桶"""
    def __init__(self, app, redis, rules: dict[str, tuple[int, int]]):
        """rules: {"/api/runs:POST": (max_count, window_seconds), ...}"""
        super().__init__(app); self._r = redis; self._rules = rules

    async def dispatch(self, request: Request, call_next):
        user = getattr(request.state, "user", None)
        if user:
            key = f"{request.url.path}:{request.method}"
            rule = self._rules.get(key)
            if rule:
                max_n, win = rule
                bucket = f"rl:{user.id}:{key}"
                c = await self._r.incr(bucket)
                if c == 1: await self._r.expire(bucket, win)
                if c > max_n: raise TooManyRequests(f"rate limit {key}")
        return await call_next(request)
```

```python
# app/utils/sanitize.py
from typing import Any
SENSITIVE_KEYS = {"authorization", "cookie", "password", "token", "api_key", "secret"}
TRUNC = 65536

def sanitize(obj: Any, max_str: int = TRUNC) -> Any:
    if isinstance(obj, dict):
        return {k: ("***" if k.lower() in SENSITIVE_KEYS else sanitize(v, max_str)) for k, v in obj.items()}
    if isinstance(obj, list):
        return [sanitize(x, max_str) for x in obj]
    if isinstance(obj, tuple):
        return tuple(sanitize(x, max_str) for x in obj)
    if isinstance(obj, bytes):
        import hashlib
        return {"__bytes__": len(obj), "sha256": hashlib.sha256(obj).hexdigest()}
    if isinstance(obj, str) and len(obj) > max_str:
        return obj[:max_str] + f"...[truncated {len(obj)-max_str} bytes]"
    return obj
```

## 2.14 AppContainer 装配骨架(后续章节逐步填充)

```python
# app/bootstrap.py
from dataclasses import dataclass
from typing import Any
from sqlalchemy.ext.asyncio import async_sessionmaker, AsyncSession, AsyncEngine
from app.config import Settings
from app.infra.db.session import create_engine, create_session_factory
from app.infra.mongo.client import create_mongo_client, get_mongo_db
from app.infra.redis import create_redis
from app.infra.logging import configure_logging
from app.settings_service import SettingsService

@dataclass(slots=True)
class AppContainer:
    settings: Settings
    db_engine: AsyncEngine
    session_factory: async_sessionmaker[AsyncSession]
    mongo_client: Any
    mongo_db: Any
    redis: Any
    settings_service: SettingsService
    # 以下字段在后续章节填入:
    llm_client: Any = None
    sandbox_facade: Any = None
    tool_registry: Any = None
    event_bus: Any = None
    trace_sink: Any = None
    pipeline: Any = None
    workflow_facade: Any = None

def build_container(settings: Settings) -> AppContainer:
    configure_logging(settings)
    engine = create_engine(settings)
    sf = create_session_factory(engine)
    mc = create_mongo_client(settings)
    mdb = get_mongo_db(mc, settings)
    redis = create_redis(settings)
    ss = SettingsService(sf, redis)
    return AppContainer(
        settings=settings, db_engine=engine, session_factory=sf,
        mongo_client=mc, mongo_db=mdb, redis=redis, settings_service=ss,
    )

async def startup(container: AppContainer) -> None:
    await container.settings_service.start()

async def shutdown(container: AppContainer) -> None:
    await container.db_engine.dispose()
    container.mongo_client.close()
    await container.redis.aclose()
```

```python
from contextlib import asynccontextmanager
from fastapi import FastAPI

@asynccontextmanager
async def lifespan(app: FastAPI):
    settings = get_settings()
    container = build_container(settings)
    app.state.container = container
    await startup(container)
    try:
        yield
    finally:
        await shutdown(container)

def create_app() -> FastAPI:
    app = FastAPI(title="Cascade Design Platform", lifespan=lifespan)

    # 中间件按注册顺序从外到里
    app.add_middleware(TraceIdMiddleware)
    app.add_middleware(AuthMiddleware)
    app.add_middleware(RateLimitMiddleware,
                       redis=None,     # lifespan 完成后绑定

                           "/api/runs:POST": (10, 60),
                       })

    register_errors(app)

    app.include_router(health.router)
    app.include_router(metrics.router)

    return app

app = create_app()
```
