# 04 · 图森林子系统(Bundle + 节点实例 + 全局边 + DAG 视图)

> **本章是对整图数据模型的落地**。读 00 章 §0.12.2 的术语层级图再读本章。核心:
> - **Bundle(大节点 / 节点集)** 是一等公民(代码层 = class/function)
> - **节点实例(小节点)** 可归属 Bundle,也可以游离
> - **边全局**,连接任意两个节点实例(可跨 Bundle);Bundle 本身不连线
> - **DAG 是视图**,不存储,从森林沿边计算出来
> - **森林 JSON 无 position**,渲染交前端 antv
>
> 依赖:01(CascadeForest / NodeInstance / Edge / Bundle 值对象、DTO、DDL、森林 JSON schema)、03(NodeTemplate)
> 交付物:Forest 解析 / 校验(环检测、边合法性、Bundle 归属合法性、孤儿检测)、访问者(Visitor)实现、拓扑迭代器、DAG 视图计算器、GraphRepo/Service/API
> 验收:
> 1. 给定非法森林 JSON(环 / 越界引用 / 边语义错 / bundle 指向不存在的 instance),`DesignValidator` 精准报错,错误信息含 instance_id / bundle_id
> 2. `TopologicalIterator` 对 100 节点森林稳定排序,O(V+E)
> 3. `DagComputeVisitor` 把森林里所有 root 沿边展开,产出若干 `DagView`(可横跨 bundle)
> 4. `POST /api/graphs/{id}/versions` 自动给每个节点实例冻结 `template_snapshot`;读取时原样返回
> 5. `tests/unit/graph/*` 全绿

```python
from abc import ABC
from typing import Any
from app.domain.graph.nodes import CascadeForest, Bundle, NodeInstance, Edge

class ForestVisitor(ABC):
    """对森林的只读遍历。默认 visit_* 是 noop;子类按需覆盖。"""
    def visit_forest(self, f: CascadeForest) -> Any:
        for b in f.bundles:        b.accept(self)
        for n in f.node_instances: n.accept(self)
        for e in f.edges:          e.accept(self)

    def visit_bundle(self, b: Bundle) -> Any: pass
    def visit_node(self, n: NodeInstance) -> Any: pass
    def visit_edge(self, e: Edge) -> Any: pass
```

```python
# app/services/forest_parser.py(运行/冻结)
from copy import deepcopy
from typing import Mapping
from app.tool_runtime.registry import ToolRegistry
from app.tool_runtime.loader import _row_to_tpl
from app.domain.tool.tool import (
    NodeTemplate, Scope, Engine, EdgeSemantic,

    CascadeForest, Bundle, NodeInstance, Edge,

from app.domain.graph.builders import TemplateResolver
from app.domain.graph.errors import GraphParseError

class RuntimeResolver(TemplateResolver):
    """保存时用:拉当前 NodeTemplate 准备冻结写回"""
    def __init__(self, registry: ToolRegistry) -> None: self._r = registry
    def resolve(self, *, template_id, version, hint_snapshot=None) -> NodeTemplate:
        import asyncio
        return asyncio.get_event_loop().run_until_complete(
            self._r.get_by_id(template_id, version)
        )

class FrozenResolver(TemplateResolver):
    """读取时用:用 hint_snapshot(即 node 里冻结的 template_snapshot)还原"""
    def resolve(self, *, template_id, version, hint_snapshot=None) -> NodeTemplate:
        if not hint_snapshot:
            raise GraphParseError(f"template_snapshot missing for {template_id}")
        return _snapshot_dict_to_template(hint_snapshot)

def _snapshot_dict_to_template(snap: Mapping) -> NodeTemplate:
    desc = "\n".join(snap["description"]) if isinstance(snap["description"], list) else snap["description"]
    return NodeTemplate(
        id=snap["id"], name=snap["name"], display_name=snap["display_name"],
        category=snap["category"], scope=Scope(snap["scope"]),
        version=snap["version"],
        description=desc,
        input_schema=snap["input_schema"], output_schema=snap["output_schema"],
        simulator=JsonSimulatorSpec(
            engine=Engine(snap["simulator"]["engine"]),
            python_impl=snap["simulator"].get("python_impl"),
            llm_fallback=bool(snap["simulator"].get("llm_fallback", False)),
        ),
        edge_semantics=tuple(EdgeSemantic(e["field"], e.get("description", ""))
                              for e in snap.get("edge_semantics", [])),
        code_hints=CodeGenerationHints(
            style_hints=tuple(snap.get("code_hints", {}).get("style_hints", [])),
            forbidden=tuple(snap.get("code_hints", {}).get("forbidden", [])),
            example_fragment=snap.get("code_hints", {}).get("example_fragment", ""),
        ),
        extensions=snap.get("extensions", {}),
        definition_hash=snap.get("definition_hash", ""),
        owner_id=snap.get("owner_id"),
    )

def _template_to_snapshot_dict(t: NodeTemplate) -> dict:
    """把 NodeTemplate 序列化回 dict,用于写入 graph_version.snapshot.node_instances[].template_snapshot"""
    return {
        "id": t.id, "name": t.name, "display_name": t.display_name,
        "category": t.category, "scope": t.scope.value, "version": t.version,
        "description": t.description.split("\n"),     # 数组形式
        "input_schema": dict(t.input_schema),
        "output_schema": dict(t.output_schema),
        "simulator": {"engine": t.simulator.engine.value,
                      "python_impl": t.simulator.python_impl,
                      "llm_fallback": t.simulator.llm_fallback},
        "edge_semantics": [{"field": e.field, "description": e.description}
                           for e in t.edge_semantics],
        "code_hints": {"style_hints": list(t.code_hints.style_hints),
                       "forbidden":   list(t.code_hints.forbidden),
                       "example_fragment": t.code_hints.example_fragment},
        "extensions": dict(t.extensions),
        "definition_hash": t.definition_hash,
        "owner_id": t.owner_id,
    }
```

### 4.4.3 Builder(从 dict 构造值对象)

```python
# app/domain/graph/builders.py(续)
from app.domain.graph.nodes import (
    CascadeForest, Bundle, NodeInstance, Edge,
)

def build_forest(
    *,
    graph_version_id: str,
    version_number: int,
    snapshot: dict,
    resolver: TemplateResolver,
) -> CascadeForest:
    """
    snapshot 结构:
      { bundles:[], node_instances:[], edges:[], metadata:{} }
    """
    try:
        bundles_src = snapshot.get("bundles", [])
        nodes_src   = snapshot["node_instances"]
        edges_src   = snapshot["edges"]
    except KeyError as e:
        raise GraphParseError(f"missing key: {e}")

    # 先算每个 instance_id 所属 bundle_id(用于 NodeInstance.bundle_id)
    iid_to_bid: dict[str, str] = {}
    for b in bundles_src:
        for iid in b.get("node_instance_ids", []):
            if iid in iid_to_bid:
                from app.domain.graph.errors import BundleMembershipConflict
                raise BundleMembershipConflict(
                    f"instance {iid} belongs to multiple bundles",
                    instance_id=iid,
                    bundles=[iid_to_bid[iid], b["bundle_id"]],
                )
            iid_to_bid[iid] = b["bundle_id"]

    nodes = tuple(_build_node(n, iid_to_bid, resolver) for n in nodes_src)
    edges = tuple(_build_edge(e) for e in edges_src)
    bundles = tuple(_build_bundle(b) for b in bundles_src)

    return CascadeForest(
        graph_version_id=graph_version_id,
        version_number=version_number,
        bundles=bundles,
        node_instances=nodes,
        edges=edges,
        metadata=snapshot.get("metadata", {}),
    )

def _build_node(n: dict, iid_to_bid: dict[str, str], resolver: TemplateResolver) -> NodeInstance:
    tpl = resolver.resolve(
        template_id=n["template_id"],
        version=n.get("template_version"),
        hint_snapshot=n.get("template_snapshot"),
    )
    return NodeInstance(
        instance_id=n["instance_id"],
        template_snapshot=tpl,
        instance_name=n.get("instance_name", n["instance_id"]),
        field_values=dict(n.get("field_values", {})),
        bundle_id=iid_to_bid.get(n["instance_id"]),
    )

def _build_edge(e: dict) -> Edge:
    return Edge(
        edge_id=e["edge_id"],
        src=e.get("from") or e["src"],
        dst=e.get("to") or e["dst"],
        semantic=e.get("edge_semantic") or e.get("semantic"),
    )

def _build_bundle(b: dict) -> Bundle:
    return Bundle(
        bundle_id=b["bundle_id"],
        name=b.get("name", b["bundle_id"]),
        description=b.get("description", ""),
        node_instance_ids=tuple(b.get("node_instance_ids", [])),
    )
```

```python
# app/services/forest_parser.py(续)
from app.domain.graph.builders import build_forest

class ForestParser:
    def __init__(self, registry: ToolRegistry) -> None:
        self._registry = registry

    async def freeze_snapshot(self, raw_snapshot: dict) -> dict:
        """保存时给每个 node_instance 填 template_snapshot(冻结当前模板)。"""
        out = deepcopy(raw_snapshot)
        for n in out.get("node_instances", []):
            tpl = await self._registry.get_by_id(n["template_id"], n.get("template_version"))
            n["template_version"] = tpl.version
            n["template_snapshot"] = _template_to_snapshot_dict(tpl)
        return out

    def parse_readonly(self, *, graph_version_id: str, version_number: int,
                       snapshot: dict) -> CascadeForest:
        """读取时用 snapshot 里冻结的 template_snapshot,不再查 Registry"""
        return build_forest(
            graph_version_id=graph_version_id,
            version_number=version_number,
            snapshot=snapshot,
            resolver=FrozenResolver(),
        )
```

### 4.5.1 CycleCheckerVisitor

> 环检测 = 看整张森林的有向图(节点实例 + 边)有没有环,**和 Bundle 归属无关**(因为 Bundle 本身不连边)。

```python
# app/domain/graph/visitors/cycle_checker.py
from app.domain.graph.visitor import ForestVisitor
from app.domain.graph.nodes import CascadeForest
from app.domain.graph.errors import GraphHasCycle

WHITE, GRAY, BLACK = 0, 1, 2

class CycleCheckerVisitor(ForestVisitor):
    def visit_forest(self, f: CascadeForest) -> None:
        nodes = {n.instance_id for n in f.node_instances}
        adj: dict[str, list[str]] = {iid: [] for iid in nodes}
        for e in f.edges:
            if e.src not in adj or e.dst not in adj:
                continue        # 会由 NodeRefCheckerVisitor 报错;此处只查环
            adj[e.src].append(e.dst)
        color = {iid: WHITE for iid in nodes}
        path: list[str] = []
        def dfs(u: str):
            color[u] = GRAY; path.append(u)
            for v in adj[u]:
                if color[v] == GRAY:
                    cycle = path[path.index(v):] + [v]
                    raise GraphHasCycle("cycle detected", path=cycle)
                if color[v] == WHITE: dfs(v)
            color[u] = BLACK; path.pop()
        for u in sorted(nodes):
            if color[u] == WHITE: dfs(u)
```

### 4.5.2 NodeRefCheckerVisitor

```python
# app/domain/graph/visitors/node_ref.py
from app.domain.graph.visitor import ForestVisitor
from app.domain.graph.errors import NodeRefInvalid

class NodeRefCheckerVisitor(ForestVisitor):
    """边/Bundle 引用的 instance_id 必须真的存在"""
    def visit_forest(self, f):
        ids = {n.instance_id for n in f.node_instances}
        # 边
        for e in f.edges:
            if e.src not in ids:
                raise NodeRefInvalid("edge src missing", edge=e.edge_id, missing=e.src)
            if e.dst not in ids:
                raise NodeRefInvalid("edge dst missing", edge=e.edge_id, missing=e.dst)
            if e.src == e.dst:
                from app.domain.graph.errors import SelfLoopEdge
                raise SelfLoopEdge("edge points to itself", edge=e.edge_id)
        # bundle
        for b in f.bundles:
            for iid in b.node_instance_ids:
                if iid not in ids:
                    raise NodeRefInvalid("bundle ref missing", bundle=b.bundle_id, missing=iid)
```

```python
from app.domain.graph.visitor import ForestVisitor
from app.domain.graph.errors import EdgeSemanticInvalid

class EdgeSemanticVisitor(ForestVisitor):
    """每条边的 semantic 必须是 src 节点模板 edge_semantics 里的某个 field"""
    def visit_forest(self, f):
        by_id = {n.instance_id: n for n in f.node_instances}

            if src is None: continue      # NodeRefChecker 会抓

                    f"edge semantic '{e.semantic}' not in {sorted(allowed)}",
                    edge=e.edge_id, src=e.src,
                )
```

```python
# app/domain/graph/visitors/orphan.py
from app.domain.graph.visitor import ForestVisitor

class OrphanReportVisitor(ForestVisitor):
    """找出:
       - 不归属任何 bundle 的游离节点(info 级,允许)
       - 既不是 root、也没有任何入边的孤立节点(warning)
    """
    def __init__(self):
        self.orphans: list[str] = []           # 游离(不在 bundle 里)
        self.isolated: list[str] = []          # 既无入也无出
    def visit_forest(self, f):
        in_deg = {n.instance_id: 0 for n in f.node_instances}
        out_deg = {n.instance_id: 0 for n in f.node_instances}
        for e in f.edges:
            if e.src in out_deg: out_deg[e.src] += 1
            if e.dst in in_deg:  in_deg[e.dst] += 1
        for n in f.node_instances:
            if n.bundle_id is None:
                self.orphans.append(n.instance_id)
            if in_deg[n.instance_id] == 0 and out_deg[n.instance_id] == 0:
                self.isolated.append(n.instance_id)
```

```python
# app/domain/graph/visitors/diff.py
from dataclasses import dataclass, field
from app.domain.graph.nodes import CascadeForest

@dataclass
class ForestDiff:
    added_bundles: list[str]     = field(default_factory=list)
    removed_bundles: list[str]   = field(default_factory=list)
    added_nodes: list[str]       = field(default_factory=list)
    removed_nodes: list[str]     = field(default_factory=list)
    changed_nodes: list[dict]    = field(default_factory=list)  # {iid, delta}
    added_edges: list[str]       = field(default_factory=list)
    removed_edges: list[str]     = field(default_factory=list)
    bundle_membership_changes: list[dict] = field(default_factory=list)

def diff(a: CascadeForest, b: CascadeForest) -> ForestDiff:
    out = ForestDiff()
    ab = {x.bundle_id for x in a.bundles}
    bb = {x.bundle_id for x in b.bundles}
    out.added_bundles   = sorted(bb - ab)
    out.removed_bundles = sorted(ab - bb)

    an = {n.instance_id: n for n in a.node_instances}
    bn = {n.instance_id: n for n in b.node_instances}
    out.added_nodes   = sorted(bn.keys() - an.keys())
    out.removed_nodes = sorted(an.keys() - bn.keys())
    for iid in an.keys() & bn.keys():
        delta = {}
        if an[iid].template_snapshot.name != bn[iid].template_snapshot.name:
            delta["template_name"] = (an[iid].template_snapshot.name, bn[iid].template_snapshot.name)
        if an[iid].template_snapshot.version != bn[iid].template_snapshot.version:
            delta["template_version"] = (an[iid].template_snapshot.version, bn[iid].template_snapshot.version)
        if dict(an[iid].field_values) != dict(bn[iid].field_values):
            delta["field_values"] = (dict(an[iid].field_values), dict(bn[iid].field_values))
        if an[iid].bundle_id != bn[iid].bundle_id:
            out.bundle_membership_changes.append({"iid": iid,
                                                   "from": an[iid].bundle_id,
                                                   "to":   bn[iid].bundle_id})
        if delta:
            out.changed_nodes.append({"iid": iid, "delta": delta})

    ae = {e.edge_id for e in a.edges}
    be = {e.edge_id for e in b.edges}
    out.added_edges   = sorted(be - ae)
    out.removed_edges = sorted(ae - be)
    return out
```

## 4.6 拓扑迭代器(森林级)

```python
# app/domain/graph/iteration.py
from typing import Iterator
from app.domain.graph.nodes import CascadeForest, NodeInstance
from app.domain.graph.errors import GraphHasCycle

class TopologicalIterator:
    """Kahn 算法。在**全森林**范围排序(Bundle 不影响顺序)。
    同一层按 instance_id 字典序,保证稳定。"""
    def __init__(self, forest: CascadeForest) -> None:
        self._f = forest
    def __iter__(self) -> Iterator[NodeInstance]:
        by_id = {n.instance_id: n for n in self._f.node_instances}
        indeg: dict[str, int] = {iid: 0 for iid in by_id}
        adj: dict[str, list[str]] = {iid: [] for iid in by_id}
        for e in self._f.edges:
            if e.src in indeg and e.dst in indeg:
                adj[e.src].append(e.dst); indeg[e.dst] += 1

        ready = sorted([iid for iid, k in indeg.items() if k == 0])
        order: list[str] = []
        while ready:
            u = ready.pop(0)
            order.append(u)
            nexts: list[str] = []
            for v in adj[u]:
                indeg[v] -= 1
                if indeg[v] == 0: nexts.append(v)
            ready = sorted(ready + nexts)
        if len(order) != len(by_id):
            raise GraphHasCycle("cycle detected during topo sort")
        return iter(by_id[i] for i in order)
```

## 4.7 DAG 视图计算器

```python
# app/domain/graph/dag_compute.py
from app.domain.graph.visitor import ForestVisitor
from app.domain.graph.nodes import CascadeForest
from app.domain.graph.dag_view import DagView

class DagComputeVisitor(ForestVisitor):
    """把森林按 root 分解成若干 DagView。
    root 定义:入度 0 的节点实例。
    一个 DAG = 从一个 root 出发沿边可达的子图(可横跨 bundle)。
    多个 DAG 之间可能共享节点(罕见但允许 v1 出现 → warning)。
    """
    def __init__(self):
        self.dags: list[DagView] = []

    def visit_forest(self, f: CascadeForest) -> None:
        by_id = {n.instance_id: n for n in f.node_instances}
        adj:    dict[str, list[str]] = {iid: [] for iid in by_id}
        edge_map: dict[tuple[str, str], list[str]] = {}
        in_deg: dict[str, int] = {iid: 0 for iid in by_id}
        for e in f.edges:
            if e.src in adj and e.dst in in_deg:
                adj[e.src].append(e.dst); in_deg[e.dst] += 1
                edge_map.setdefault((e.src, e.dst), []).append(e.edge_id)

        roots = sorted(iid for iid, k in in_deg.items() if k == 0)
        for idx, root in enumerate(roots):
            nodes, edges = _reach(root, adj, edge_map)
            spans = sorted({by_id[i].bundle_id for i in nodes if by_id[i].bundle_id})
            self.dags.append(DagView(
                dag_index=idx, root=root,
                node_ids=tuple(sorted(nodes)),
                edge_ids=tuple(sorted(edges)),
                spans_bundles=tuple(spans),
            ))

def _reach(root, adj, edge_map) -> tuple[set[str], set[str]]:
    seen_n: set[str] = set()
    seen_e: set[str] = set()
    stack = [root]
    while stack:
        u = stack.pop()
        if u in seen_n: continue
        seen_n.add(u)
        for v in adj.get(u, []):
            for eid in edge_map.get((u, v), []): seen_e.add(eid)
            if v not in seen_n: stack.append(v)
    return seen_n, seen_e
```

```python
# app/services/design_validator.py
from dataclasses import dataclass, field
from app.domain.graph.nodes import CascadeForest
from app.domain.graph.visitors.cycle_checker      import CycleCheckerVisitor
from app.domain.graph.visitors.node_ref           import NodeRefCheckerVisitor

from app.domain.graph.visitors.duplicate_edge     import DuplicateEdgeVisitor
from app.domain.graph.visitors.orphan             import OrphanReportVisitor
from app.domain.errors import BusinessError

@dataclass
class ValidationReport:
    errors: list[dict] = field(default_factory=list)
    warnings: list[dict] = field(default_factory=list)
    @property
    def ok(self) -> bool: return not self.errors

class DesignValidator:
    """**被 07 章 Phase1 Handler 1 调用**。
    错误顺序:严重错(环 / 引用无效) short-circuit;其余并行收集。"""

    def run(self, forest: CascadeForest) -> ValidationReport:
        rep = ValidationReport()
        # 严重错:环 / 引用无效 → 立即返回
        try:    NodeRefCheckerVisitor().visit_forest(forest)
        except BusinessError as e: rep.errors.append(_err(e)); return rep
        try:    CycleCheckerVisitor().visit_forest(forest)
        except BusinessError as e: rep.errors.append(_err(e)); return rep
        # 可并列:边语义 + 字段 schema + 重复边
        try:    EdgeSemanticVisitor().visit_forest(forest)
        except BusinessError as e: rep.errors.append(_err(e))
        try:    DuplicateEdgeVisitor().visit_forest(forest)
        except BusinessError as e: rep.errors.append(_err(e))
        try:    SchemaValidationVisitor().visit_forest(forest)
        except BusinessError as e: rep.errors.append(_err(e))
        # 报告类:孤立节点 / 游离节点
        orphan = OrphanReportVisitor()
        orphan.visit_forest(forest)
        for iid in orphan.isolated:
            rep.warnings.append({"code": "ISOLATED_NODE", "instance_id": iid})
        for iid in orphan.orphans:
            rep.warnings.append({"code": "BUNDLE_FREE_NODE", "instance_id": iid})
        return rep

def _err(e: BusinessError) -> dict:
    return {"code": e.code, "message": e.message, "extra": e.extra}
```

```python
# app/domain/graph/paste.py
"""前端 ctrl+c ctrl+v 支持。前端本来就能自己 copy JSON,但边界情况(id 冲突、跨图 template_id 不存在)由本模块提供一层服务端兜底。

v1 后端无独立 API,纯工具函数。前端可以直接本地重发 id 后 PUT /api/graphs/{id}/draft。
v2 可以加 POST /api/graphs/{id}/paste 做更严谨的服务端校验。"""

import uuid
from typing import Iterable

def rebuild_ids(pasted: dict) -> dict:
    """输入 { bundles, node_instances, edges } 片段,重新生成所有 bundle/instance/edge id。
    返回新片段。内部边/bundle 引用自动跟着改。"""
    id_map: dict[str, str] = {}

    def new_iid(old: str) -> str:
        if old in id_map: return id_map[old]
        prefix = old.split("_", 1)[0]
        new = f"{prefix}_{uuid.uuid4().hex[:8]}"
        id_map[old] = new
        return new

    for n in pasted.get("node_instances", []):
        n["instance_id"] = new_iid(n["instance_id"])
    for b in pasted.get("bundles", []):
        b["bundle_id"] = new_iid(b["bundle_id"])
        b["node_instance_ids"] = [id_map.get(x, x) for x in b.get("node_instance_ids", [])]
    for e in pasted.get("edges", []):
        e["edge_id"] = new_iid(e["edge_id"])
        e["from"] = id_map.get(e.get("from") or e.get("src"), e.get("from") or e.get("src"))
        e["to"]   = id_map.get(e.get("to")   or e.get("dst"), e.get("to")   or e.get("dst"))
    return pasted
```

```python
class GraphVersionRepo(ABC):
    @abstractmethod
    async def save_new(self, graph_id: str, snapshot: dict, commit_message: str,
                       parent_version_id: str | None, user_id: int) -> GraphVersionRow: ...
    @abstractmethod
    async def get(self, version_id: str) -> GraphVersionRow: ...
    @abstractmethod
    async def get_by_number(self, graph_id: str, version_number: int) -> GraphVersionRow: ...
    @abstractmethod
    async def list(self, graph_id: str) -> list[GraphVersionRow]: ...

class SqlGraphVersionRepo(SqlRepoBase, GraphVersionRepo):
    async def save_new(self, graph_id, snapshot, commit_message, parent_version_id, user_id):
        latest = (await self._s.execute(
            select(GraphVersionRow)
              .where(GraphVersionRow.graph_id == graph_id)
              .order_by(GraphVersionRow.version_number.desc())
        )).scalars().first()
        if latest is not None and parent_version_id is not None and latest.id != parent_version_id:
            raise StaleVersion("graph version changed",
                              expected=parent_version_id, latest=latest.id)
        new_num = (latest.version_number if latest else 0) + 1
        vid = new_id("gv")
        self._s.add(GraphVersionRow(
            id=vid, graph_id=graph_id, version_number=new_num, snapshot=snapshot,
            commit_message=commit_message, parent_version_id=(latest.id if latest else None),
            created_by=user_id,
        ))
        await self._s.flush()
        await self._s.execute(
            update(CascadeGraphRow).where(CascadeGraphRow.id == graph_id).values(latest_version_id=vid)
        )
        r = (await self._s.execute(
            select(GraphVersionRow).where(GraphVersionRow.id == vid)
        )).scalar_one()
        return r
    async def get(self, version_id):
        r = (await self._s.execute(
            select(GraphVersionRow).where(GraphVersionRow.id == version_id)
        )).scalar_one_or_none()
        if r is None: raise NotFound(version_id)
        return r
    async def get_by_number(self, graph_id, version_number):
        r = (await self._s.execute(
            select(GraphVersionRow).where(
                GraphVersionRow.graph_id == graph_id,
                GraphVersionRow.version_number == version_number,
            )
        )).scalar_one_or_none()
        if r is None: raise NotFound(f"{graph_id}@{version_number}")
        return r
    async def list(self, graph_id):
        return list((await self._s.execute(
            select(GraphVersionRow)
              .where(GraphVersionRow.graph_id == graph_id)
              .order_by(GraphVersionRow.version_number.desc())
        )).scalars().all())
```

- 合法森林 → ok=True, warnings 可以非空(游离节点 info)
- 简单环 → errors 中只有 `VALIDATION_GRAPH_HAS_CYCLE`,不再跑下去
- 合法结构 + 字段值非法 → errors 中是 `VALIDATION_FIELD_VALUE_INVALID`;orphan warning 仍产生

### 4.14.5 ForestParser

- `freeze_snapshot` 后每个节点含 `template_snapshot` 且 `definition_hash` 与 DB 一致
- `parse_readonly` 不依赖 Registry:临时删掉对应模板后仍能解析老版本

### 4.14.6 集成 API

- 建图 → 保存 v1 → 列版本 = 1 → 编辑 snapshot → 保存 v2(parent_version_id = v1 的 id)= 2 条
- 保存时 parent_version_id 错 → 409 StaleVersion
- Draft 并发写:后写覆盖前写
- `_validate` 对非法森林返回详细 errors(不入库)

### 4.14.7 粘贴

- `rebuild_ids` 对一个包含 bundle + nodes + edges 的片段,所有 id 重发;内部引用全部跟着改;重复调用产生不同 id
